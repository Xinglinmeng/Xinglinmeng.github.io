public class Singleton {//双重检查锁
  // volatile确保多线程正确处理instance变量
  private static volatile Singleton instance;
  // 私有构造函数，防止外部实例化
  private Singleton() {}
  // 提供全局访问点，双重检查锁优化性能
  public static Singleton getInstance() {
   if (instance == null) {
     synchronized (Singleton.class) {
       if (instance == null) {
         instance = new Singleton();
       }
     }
   }
   return instance;
  }
}

public class ReverseString {
  public static void main(String[] args) {
   String original = "Hello, World!";
   char[] charArray = original.toCharArray();
   int left = 0;
   int right = charArray.length - 1;
   while (left < right) {
     // 交换字符
     char temp = charArray[left];
     charArray[left] = charArray[right];
     charArray[right] = temp;
     left++;
     right--;
   }

   String reversed = new String(charArray);
   System.out.println(reversed); 
  }
}//若在六行内实现可以for循环中初始化，下标移动

6.Java二分数据检索，插入

public class BinarySearch {
  public static int binarySearch(int[] array, int target) {
   int left = 0;
   int right = array.length - 1;                       
   while (left <= right) {
     int mid = left + (right - left)/2; // 防止(left + right)可能导致的溢出
     if (array[mid] == target) {
       return mid; // 找到目标值，返回索引
     } else if (array[mid] < target) {
       left = mid + 1; // 在右半部分继续查找
     } else {
       right = mid - 1; // 在左半部分继续查找
     }
   }
   return -1; // 未找到目标值，返回-1
  }

  public static void main(String[] args) {
   int[] sortedArray = {1, 3, 5, 7, 9, 11, 13};
   int target = 7;
   int result = binarySearch(sortedArray, target);
   if (result != -1) {
     System.out.println("Found " + target + " at index " + result);
   } else {
     System.out.println(target + " not found in the array.");
   }
  }
}
public class BinaryInsertion {
  public static void binaryInsert(int[] array, int key) {
  int left = 0, right = array.length - 1;
  int insertPos = array.length; // 默认插入位置为数组末尾，如果二分查找没有找到合适的位置，则插入末尾（但这里我们会找到合适位置并更新它）
  // 二分查找插入位置
  while (left <= right) {
    int mid = left + (right - left) / 2;
    if (array[mid] > key) {
      insertPos = mid; // 更新插入位置为mid（实际上应该插入mid+1的位置，但这里先记录mid，稍后处理）
      right = mid - 1;
    } else {
      left = mid + 1;
    }
  }
  // 注意：由于我们希望在相等元素之后插入以保持排序稳定性，因此insertPos不需要-1（它已经是mid之后的位置）
  // 但由于数组是从0开始索引的，并且我们是在找到比key大的元素时停止的，所以insertPos实际上是正确的插入点（即第一个大于key的元素的位置）
  // 将数组元素向右移动以腾出空间
  for (int i = array.length - 1; i >= insertPos; i--) {
    array[i + 1] = array[i];
  }
  // 插入key
  array[insertPos] = key;
  }

  public static void main(String[] args) {
  int[] sortedArray = {1, 3, 5, 7, 9};
  int keyToInsert = 6;
  System.out.println("Before insertion: " + Arrays.toString(sortedArray));
  binaryInsert(sortedArray, keyToInsert);
  System.out.println("After insertion: " + Arrays.toString(sortedArray));
  }
}

7.TCP和UDP(没问过，随手记的)

TCP（Transmission Control Protocol，传输控制协议）

UDP（User Datagram Protocol，用户数据报协议）是两种常用的数据传输协议

TCP协议的特点

  1. 面向连接：TCP在传输数据之前，需要先建立连接。这种连接是全双工的，即数据可以在两个方向上同时传输。连接建立通过三次握手过程完成，而连接关闭则通过四次挥手过程实现。
  2. 可靠性：TCP通过一系列机制确保数据的可靠传输，包括序号机制、确认应答机制、超时重传机制等。如果发送方在合理的往返时延内未收到接收方的确认应答，则会重传对应的报文段。
  3. 基于字节流：TCP将应用层发送的数据视为无结构的字节流，并按照字节流的方式传输数据。
  4. 流量控制与拥塞控制：TCP通过滑动窗口机制实现流量控制，以防止发送方发送过多的数据导致接收方无法处理。同时，TCP还包含拥塞控制机制，以避免网络拥塞，确保网络资源的合理利用。

TCP的应用场景：由于TCP提供可靠的数据传输和流量控制机制，因此适用于要求高可靠性的应用，如网页浏览、文件传输、电子邮件等。在这些应用中，数据的完整性和顺序性至关重要。

UDP协议的特点

  1. 无连接：UDP在传输数据之前不需要建立连接，因此传输速度较快。同时，由于没有连接状态需要维护，UDP的头部开销也较小。
  2. 不可靠性：UDP不提供数据包的重传、错误校验等机制，因此不保证数据的可靠送达。数据包可能会丢失、重复或乱序。
  3. 实时性：由于无需进行握手和确认机制，UDP的传输速度较快，适用于实时性要求较高的场景。
  4. 头部开销小：UDP的头部仅有8个字节，远小于TCP的头部开销。

UDP的应用场景：UDP则适用于对速度和效率要求更高的环境，如视频流、在线游戏和语音传输等实时应用。这些应用更注重传输速度而不是数据包的完整性，因此可以容忍少量的数据包丢失或乱序。

public class CustomHash {
  public static void main(String[] args) {
  char[] chars = {'a', 'b', 'c'};
  int[] hashes = new int[chars.length];
  // 自定义哈希函数：ASCII \ 31 + 固定值
  for (int i = 0; i < chars.length; i++) {
    int hash = chars[i] \* 31 + 7; // 示例哈希算法
    hashes[i] = hash;
  }
  for (int hash : hashes) {
    System.out.print(hash + " ");
  }
  // 输出：3014 3045 3076
  }
}

12.三个线程，两个同时执行，执行完后，第三个线程启动

public class ThreadSyncExample {
  public static void main(String[] args) throws InterruptedException {
   // 创建并启动线程1和线程2
   Thread thread1 = new Thread(() -> {
     System.out.println("线程1开始执行");
     // 模拟耗时操作
     try { 
         Thread.sleep(1000); 
         } catch (InterruptedException e) {
     }
     System.out.println("线程1执行完成");
   });
      
   Thread thread2 = new Thread(() -> {
     System.out.println("线程2开始执行");
     // 模拟耗时操作
     try { Thread.sleep(1500); 
         } catch (InterruptedException e) {
     }
     System.out.println("线程2执行完成");
   });
      
   // 同时启动前两个线程
   thread1.start();
   thread2.start();
      
   // 等待两个线程都执行完成
   thread1.join();
   thread2.join();
      
   // 创建并启动线程3
   Thread thread3 = new Thread(() -> {
     System.out.println("线程3开始执行");
     System.out.println("线程3执行完成");
   });
   thread3.start();
  }
}

20.Mysq索引什么情况会失效、怎么判断索引失效

常见导致索引失效的场景包括：

  1. 对索引字段进行函数或表达式运算

    示例：WHERE YEAR(create_time) = 2023
    对索引字段 create_time 使用函数 YEAR() 后，索引无法直接生效，需改为 WHERE create_time BETWEEN '2023-01-01' AND '2023-12-31'。

  2. 隐式类型转换

    示例：索引字段为 INT 类型，查询条件为 WHERE id = '123'（字符串类型）
    MySQL 会将 '123' 转换为整数 123，导致索引失效。需确保查询条件与字段类型一致。

  3. 使用 LIKE 模糊查询且通配符在前

    示例：WHERE name LIKE '%abc%'
    通配符 % 在前会导致全表扫描，无法利用索引。若查询前缀（如 WHERE name LIKE 'abc%'），则索引可生效。

  4. 使用 OR 连接非索引字段

    示例：WHERE id = 123 OR name = 'test'
    若 name 字段无索引，则 OR 条件会触发全表扫描。需确保 OR 两侧字段均有索引。

  5. 范围查询后字段失效

    示例：WHERE age > 30 AND name = 'test'
    若索引为 (age, name)，范围查询 age > 30 后，name 的索引将失效。

  6. 查询条件包含 IS NULL 或 IS NOT NULL

    示例：WHERE name IS NULL
    除非索引字段明确支持 NULL 值，否则索引可能失效。

  7. 不等于（!=）或 NOT IN 查询

    示例：WHERE age != 18 或 WHERE age NOT IN (18, 19)
    此类查询通常导致全表扫描。

  8. 低选择性字段索引

    示例：对性别字段（gender）建立索引，但表中男女比例接近 1:1，索引选择性低，优化效果有限。

  9. 使用 ORDER BY 或 GROUP BY 未遵循索引顺序

    示例：索引为 (a, b)，但查询为 ORDER BY b, a，索引无法完全利用。

  10. 统计信息未更新

    示例：表中数据大幅更新后，未执行 ANALYZE TABLE 更新统计信息，导致优化器选择次优执行计划。

21.集合相关的接口以及实现类

Collection 接口体系

  1. List 接口
  • 特点：有序、可重复，允许索引访问。
  • 典型实现类
    • ArrayList：基于动态数组实现，适合随机访问（get(index)），但插入/删除效率较低。
    • LinkedList：基于双向链表实现，适合频繁插入/删除操作，但随机访问效率低。
    • Vector：线程安全版本，性能较 ArrayList 低，已被 ArrayList 替代。
    • Stack：继承自 Vector，提供栈操作（如 push()、pop()），但已不推荐使用。
  1. Set 接口
  • 特点：无序、不可重复。
  • 典型实现类
    • HashSet：基于哈希表实现，元素无序，依赖 hashCode() 和 equals() 方法判断重复。
    • LinkedHashSet：HashSet 的链表版本，维护插入顺序。
    • TreeSet：基于红黑树实现，元素自然排序或通过比较器排序。

Map 接口

  • 特点：键值对存储，键唯一，值可重复。
  • 典型实现类
    • HashMap：基于哈希表实现，键无序，允许一个 null 键和多个 null 值。
    • LinkedHashMap：HashMap 的链表版本，维护插入顺序或访问顺序。
    • TreeMap：基于红黑树实现，键自然排序或通过比较器排序。
    • Hashtable：线程安全版本，性能较低，已被 ConcurrentHashMap 替代。
    • ConcurrentHashMap：线程安全，支持高并发访问，性能优于 Hashtable。

  2. 查询缓存：缓存查询结果，但可能已经被废弃。
  3. 解析器：进行语法分析，生成解析树。
  4. 预处理器：检查语义，如表是否存在，权限验证。
  5. 优化器：生成执行计划。
  6. 执行器：执行查询，调用存储引擎接口。
  7. 存储引擎：实际读写数据。

事务相关的操作可能还涉及到日志模块，比如redo log、undo log和binlog。比如，在更新数据的时候，会先写redo log，再写binlog，最后提交事务的时候确保日志持久化。不过这可能属于写入过程的步骤，查询的话可能不需要这些。

26.Union和union all

UNION：合并多个查询结果集，并自动去除重复记录，最终返回的结果集是去重后的集合。由于需要去重，数据库引擎必须对结果集进行排序和比较，因此性能较低，尤其是当数据量较大时。

UNION ALL：合并多个查询结果集，保留所有记录（包括重复记录），最终返回的结果集是原始集合的拼接。直接拼接结果集，无需去重操作，性能更高，适合需要保留重复记录的场景。

27.事务传播

当事务方法被另一个事务方法调用时，事务应如何处理

传播行为 描述 案例
REQUIRED 如果当前存在事务，则加入该事务；如果当前没有事务，则创建一个新事务。 默认传播行为，适用于大多数场景。例如，用户下单时，订单服务和库存服务需要共享同一个事务。
REQUIRES_NEW 总是创建一个新事务，并挂起当前事务（如果存在）。 例如，记录日志时，日志服务需要独立于主业务事务，确保日志始终被记录，即使主业务事务回滚。
SUPPORTS 如果当前存在事务，则加入该事务；如果当前没有事务，则以非事务方式执行。 适用于可选事务场景。例如，查询用户信息时，如果存在事务则加入，否则直接查询。
NOT_SUPPORTED 以非事务方式执行，并挂起当前事务（如果存在）。 适用于不需要事务的场景。例如，导出报表时，为了避免事务占用资源，可以选择非事务方式执行。
MANDATORY 如果当前存在事务，则加入该事务；如果当前没有事务，则抛出异常。 适用于必须依赖外部事务的场景。例如，转账操作中，扣款和入账必须共享同一个事务，否则会导致数据不一致。
NEVER 以非事务方式执行，如果当前存在事务，则抛出异常。 适用于绝对不能有事务的场景。例如，某些敏感操作（如审计日志）需要确保没有事务干扰。
NESTED 如果当前存在事务，则创建一个嵌套事务；如果当前没有事务，则行为与 REQUIRED 相同。嵌套事务可以独立回滚，而不会影响外部事务。

API的全称是应用程序编程接口（Application Programming Interface）。它是一组预定义的函数和协议，允许不同的软件应用程序进行交互。API的主要目的是使软件开发者能够使用一些现成的代码，从而简化开发过程，提高效率和安全性（用于实现应用程序之间的数据交互和功能调用）

29.Arraylist和linkedlist数据底层是怎么存储的

  1. 元素在内存中是连续存储的，每个元素占据固定的内存空间。
  • 节点在内存中是分散存储的，每个节点占用独立的内存空间。
  • 节点之间通过指针关联，指针存储的是其他节点的内存地址。

30.Mysql存储引擎以及介绍

存储引擎是数据库底层用于存储、检索和管理数据的关键组件。不同的存储引擎提供了不同的特性，适用于不同的应用场景。

-InnoDB

  • 默认引擎：从MySQL 5.5版本开始，InnoDB成为默认存储引擎。
  • 特点
    • 事务支持：支持ACID（原子性、一致性、隔离性、持久性）事务。
    • 行级锁定：提高并发性能，减少锁冲突。
    • 外键约束：支持外键，保证数据完整性。
    • 崩溃恢复：提供崩溃恢复能力，确保数据一致性。
    • MVCC（多版本并发控制）：支持高并发读写。
  • 适用场景
    • 需要事务支持的应用（如金融系统、电商系统）。
    • 高并发读写场景。
    • 需要数据完整性和一致性的应用。

-MyISAM

  • 特点

    • 非事务型：不支持事务和外键。
    • 表级锁定：读写时会对整个表加锁，并发性能较低。
    • 全文索引：支持全文索引，适合文本搜索。
    • 高读取性能：对于只读或读多写少的场景，性能优于InnoDB。

  • 适用场景

    • 读多写少的应用（如数据仓库、日志系统）。
    • 需要全文索引的场景。

-MEMORY

  • 特点
    • 内存存储：数据存储在内存中，读写速度极快。
    • 表级锁定：并发性能较低。
    • 数据易失性：服务器重启或崩溃时，数据会丢失。
    • 固定长度行：不支持可变长度字段（如VARCHAR）。
  • 适用场景
    • 临时数据存储（如缓存、会话数据）。
    • 需要快速读写的临时表。

-CSV

  • 特点
    • CSV文件存储：数据存储在CSV文件中，便于与其他系统交换数据。
    • 不支持索引：查询性能较低。
    • 简单易用：适合小型数据集。
  • 适用场景
    • 数据导入导出。
    • 与其他系统（如Excel）交互。

-ARCHIVE

  • 特点
    • 高压缩比：数据存储高度压缩，节省存储空间。
    • 只支持插入和选择：不支持更新和删除。
    • 高写入性能：适合日志归档。
  • 适用场景
    • 日志归档和历史数据存储。

-NDB（Cluster）

  • 特点
    • 分布式存储：支持分布式集群，提供高可用性和可扩展性。
    • 同步复制：数据在多个节点间同步复制，确保数据一致性。
    • 内存存储：数据主要存储在内存中，读写性能高。
  • 适用场景
    • 高可用性和可扩展性要求高的应用（如大型网站、在线游戏）。

-FEDERATED

  • 特点

    • 联邦存储：允许访问远程MySQL服务器上的表，就像访问本地表一样。
    • 数据不存储在本地：仅存储表的元数据。

  • 适用场景

    • 分布式数据库系统。
    • 跨数据库的数据集成。

  • BLACKHOLE

  • 特点

    • 黑洞存储：写入的数据会被丢弃，不存储在任何地方。
    • 复制功能：常用于主从复制中的中继日志。

  • 适用场景

    • 数据复制的中继。
    • 测试和开发环境。

-MERGE

  • 特点
    • 表合并：将多个相同的MyISAM表合并为一个逻辑表。
    • 统一视图：对用户提供一个统一的视图。
  • 适用场景
    • 分区表管理。
    • 历史数据归档。

如何选择存储引擎？

  1. 事务支持：如果需要事务支持，选择InnoDB。
  2. 并发性能：高并发读写场景选择InnoDB，低并发读多写少场景选择MyISAM。
  3. 数据持久性：如果数据需要持久化，避免使用MEMORY引擎。
  4. 特殊需求
    • 全文索引：选择MyISAM。
    • 日志归档：选择ARCHIVE。
    • 分布式存储：选择NDB。

42.软件测试标准、目的

软件测试目的

  1. 缺陷发现

    • 识别代码、逻辑或设计中的错误，预防上线后故障。

  3. 需求验证

    • 确保软件符合功能与非功能需求（如性能、安全性）。

  5. 质量提升

    • 通过持续测试优化稳定性、用户体验及兼容性。

  7. 风险控制

    • 降低因软件缺陷导致的经济、安全或声誉损失。

  9. 合规保障

    • 满足法规（GDPR、HIPAA）及行业标准（PCI-DSS）。

  11. 过程改进

    • 分析缺陷根源，优化开发流程，减少未来错误。

  13. 性能评估

    • 测试负载、压力下的表现，确保高并发或大数据处理能力。

  15. 用户信心建立

    • 通过全面测试增强客户对产品质量的信任。
