  private Singleton() {}

   if (instance == null) {

       }
     }
   }
   return instance;
  }
}

public class ReverseString {
  public static void main(String[] args) {
   String original = "Hello, World!";
   char[] charArray = original.toCharArray();
   int left = 0;
   int right = charArray.length - 1;
   while (left < right) {
     // 交换字符
     char temp = charArray[left];
     charArray[left] = charArray[right];
     charArray[right] = temp;
     left++;
     right--;
   }

   String reversed = new String(charArray);
   System.out.println(reversed); 
  }
}//若在六行内实现可以for循环中初始化，下标移动

6.Java二分数据检索，插入

public class BinarySearch {
  public static int binarySearch(int[] array, int target) {
   int left = 0;
   int right = array.length - 1;                       
   while (left <= right) {
     int mid = left + (right - left)/2; // 防止(left + right)可能导致的溢出
     if (array[mid] == target) {
       return mid; // 找到目标值，返回索引
     } else if (array[mid] < target) {
       left = mid + 1; // 在右半部分继续查找
     } else {
       right = mid - 1; // 在左半部分继续查找
     }
   }
   return -1; // 未找到目标值，返回-1
  }

  public static void main(String[] args) {
   int[] sortedArray = {1, 3, 5, 7, 9, 11, 13};
   int target = 7;
   int result = binarySearch(sortedArray, target);
   if (result != -1) {
     System.out.println("Found " + target + " at index " + result);
   } else {
     System.out.println(target + " not found in the array.");
   }
  }
}
public class BinaryInsertion {
  public static void binaryInsert(int[] array, int key) {
  int left = 0, right = array.length - 1;
  int insertPos = array.length; // 默认插入位置为数组末尾，如果二分查找没有找到合适的位置，则插入末尾（但这里我们会找到合适位置并更新它）
  // 二分查找插入位置
  while (left <= right) {
    int mid = left + (right - left) / 2;
    if (array[mid] > key) {
      insertPos = mid; // 更新插入位置为mid（实际上应该插入mid+1的位置，但这里先记录mid，稍后处理）
      right = mid - 1;
    } else {
      left = mid + 1;
    }
  }
  // 注意：由于我们希望在相等元素之后插入以保持排序稳定性，因此insertPos不需要-1（它已经是mid之后的位置）
  // 但由于数组是从0开始索引的，并且我们是在找到比key大的元素时停止的，所以insertPos实际上是正确的插入点（即第一个大于key的元素的位置）
  // 将数组元素向右移动以腾出空间
  for (int i = array.length - 1; i >= insertPos; i--) {
    array[i + 1] = array[i];
  }
  // 插入key
  array[insertPos] = key;
  }

  public static void main(String[] args) {
  int[] sortedArray = {1, 3, 5, 7, 9};
  int keyToInsert = 6;
  System.out.println("Before insertion: " + Arrays.toString(sortedArray));
  binaryInsert(sortedArray, keyToInsert);
  System.out.println("After insertion: " + Arrays.toString(sortedArray));
  }
}

7.TCP和UDP(没问过，随手记的)

TCP（Transmission Control Protocol，传输控制协议）

UDP（User Datagram Protocol，用户数据报协议）是两种常用的数据传输协议

TCP协议的特点

  1. 面向连接：TCP在传输数据之前，需要先建立连接。这种连接是全双工的，即数据可以在两个方向上同时传输。连接建立通过三次握手过程完成，而连接关闭则通过四次挥手过程实现。
  2. 可靠性：TCP通过一系列机制确保数据的可靠传输，包括序号机制、确认应答机制、超时重传机制等。如果发送方在合理的往返时延内未收到接收方的确认应答，则会重传对应的报文段。
  3. 基于字节流：TCP将应用层发送的数据视为无结构的字节流，并按照字节流的方式传输数据。
  4. 流量控制与拥塞控制：TCP通过滑动窗口机制实现流量控制，以防止发送方发送过多的数据导致接收方无法处理。同时，TCP还包含拥塞控制机制，以避免网络拥塞，确保网络资源的合理利用。

TCP的应用场景：由于TCP提供可靠的数据传输和流量控制机制，因此适用于要求高可靠性的应用，如网页浏览、文件传输、电子邮件等。在这些应用中，数据的完整性和顺序性至关重要。

UDP协议的特点

  1. 无连接：UDP在传输数据之前不需要建立连接，因此传输速度较快。同时，由于没有连接状态需要维护，UDP的头部开销也较小。
  2. 不可靠性：UDP不提供数据包的重传、错误校验等机制，因此不保证数据的可靠送达。数据包可能会丢失、重复或乱序。
  3. 实时性：由于无需进行握手和确认机制，UDP的传输速度较快，适用于实时性要求较高的场景。
  4. 头部开销小：UDP的头部仅有8个字节，远小于TCP的头部开销。

UDP的应用场景：UDP则适用于对速度和效率要求更高的环境，如视频流、在线游戏和语音传输等实时应用。这些应用更注重传输速度而不是数据包的完整性，因此可以容忍少量的数据包丢失或乱序。

public class CustomHash {
  public static void main(String[] args) {
  char[] chars = {'a', 'b', 'c'};
  int[] hashes = new int[chars.length];
  // 自定义哈希函数：ASCII \ 31 + 固定值
  for (int i = 0; i < chars.length; i++) {
    int hash = chars[i] \* 31 + 7; // 示例哈希算法

11.快速排序

核心思想是通过一次“划分”操作将数组分为两部分，使得左边的元素都小于等于某个“基准值”（pivot），右边的元素都大于等于该基准值，然后递归地对左右两部分进行排序。

public class QuickSort {
    // 主方法，测试快速排序
    public static void main(String[] args) {
        int[] arr = {8, 3, 1, 7, 0, 10, 2};
        System.out.println("排序前：");
        printArray(arr);

        quickSort(arr, 0, arr.length - 1);

        System.out.println("排序后：");
        printArray(arr);
    }
    
    // 快速排序主函数
    public static void quickSort(int[] arr, int left, int right) {
        if (left < right) {
            // 1. 找到划分点
            int pivotIndex = partition(arr, left, right);
            // 2. 对左右两部分递归排序
            quickSort(arr, left, pivotIndex - 1);
            quickSort(arr, pivotIndex + 1, right);
        }
    }

    // 划分函数：将数组分为两部分，并返回基准值的最终位置
    public static int partition(int[] arr, int left, int right) {
        // 选择最右边的元素作为基准值
        int pivot = arr[right];
        int i = left - 1; // i是小于等于pivot的区域的最后一个元素索引

        for (int j = left; j < right; j++) {
            if (arr[j] <= pivot) {
                i++;
                // 交换arr[i]和arr[j]，将小于等于pivot的元素移到左边
                swap(arr, i, j);
            }
        }
        // 最后将pivot放到正确的位置（i+1）
        swap(arr, i + 1, right);
        return i + 1; // 返回基准值的索引
    }
    // 辅助函数：交换数组中的两个元素
    public static void swap(int[] arr, int i, int j) {
        int temp = arr[i];
        arr[i] = arr[j];
        arr[j] = temp;
    }

    // 辅助函数：打印数组
    public static void printArray(int[] arr) {
        for (int num : arr) {
            System.out.print(num + " ");
        }
        System.out.println();
    }
}

12.三个线程，两个同时执行，执行完后，第三个线程启动

public class ThreadSyncExample {
  public static void main(String[] args) throws InterruptedException {
   // 创建并启动线程1和线程2
   Thread thread1 = new Thread(() -> {
     System.out.println("线程1开始执行");
     // 模拟耗时操作
     try { 
         Thread.sleep(1000); 
         } catch (InterruptedException e) {
     }
     System.out.println("线程1执行完成");
   });
      
   Thread thread2 = new Thread(() -> {
     System.out.println("线程2开始执行");
     // 模拟耗时操作
     try { Thread.sleep(1500); 
         } catch (InterruptedException e) {
     }
     System.out.println("线程2执行完成");
   });
      
   // 同时启动前两个线程
   thread1.start();
   thread2.start();
      
   // 等待两个线程都执行完成
   thread1.join();
   thread2.join();
      
   // 创建并启动线程3
   Thread thread3 = new Thread(() -> {
     System.out.println("线程3开始执行");
     System.out.println("线程3执行完成");
   });
   thread3.start();
  }
}

Java中的AtomicInteger等原子类，利用CAS（Compare and Swap）操作来保证线程安全。然后是无锁数据结构，比如使用线程本地存储（ThreadLocal）来避免共享，或者不可变对象，因为不可变对象的状态无法改变，自然线程安全。

20.Mysq索引什么情况会失效、怎么判断索引失效

常见导致索引失效的场景包括：

  1. `对索引字段进行函数或表达式运算

    示例：WHERE YEAR(create_time) = 2023
    对索引字段 create_time 使用函数 YEAR() 后，索引无法直接生效，需改为 WHERE create_time BETWEEN '2023-01-01' AND '2023-12-31'。

  2. 隐式类型转换

    示例：索引字段为 INT 类型，查询条件为 WHERE id = '123'（字符串类型）
    MySQL 会将 '123' 转换为整数 123，导致索引失效。需确保查询条件与字段类型一致。

  3. 使用 LIKE 模糊查询且通配符在前

    示例：WHERE name LIKE '%abc%'
    通配符 % 在前会导致全表扫描，无法利用索引。若查询前缀（如 WHERE name LIKE 'abc%'），则索引可生效。

  4. 使用 OR 连接非索引字段

    示例：WHERE id = 123 OR name = 'test'
    若 name 字段无索引，则 OR 条件会触发全表扫描。需确保 OR 两侧字段均有索引。

  5. 范围查询后字段失效

    示例：WHERE age > 30 AND name = 'test'
    若索引为 (age, name)，范围查询 age > 30 后，name 的索引将失效。

  6. 查询条件包含 IS NULL 或 IS NOT NULL

    示例：WHERE name IS NULL
    除非索引字段明确支持 NULL 值，否则索引可能失效。

  7. 不等于（!=）或 NOT IN 查询

    示例：WHERE age != 18 或 WHERE age NOT IN (18, 19)
    此类查询通常导致全表扫描。

  8. 低选择性字段索引

    示例：对性别字段（gender）建立索引，但表中男女比例接近 1:1，索引选择性低，优化效果有限。

  9. 使用 ORDER BY 或 GROUP BY 未遵循索引顺序

    示例：索引为 (a, b)，但查询为 ORDER BY b, a，索引无法完全利用。

  10. 统计信息未更新

    示例：表中数据大幅更新后，未执行 ANALYZE TABLE 更新统计信息，导致优化器选择次优执行计划。

使用 EXPLAIN 分析查询计划

EXPLAIN SELECT * FROM users WHERE age > 20 AND name = 'Alice';
  • type：如果显示 ALL，表示全表扫描，索引失效。
  • key：如果为空，表示未使用索引。
  • possible_keys：显示可能使用的索引。
  • rows：扫描的行数，行数越多，性能越差。

查看查询性能：使用 SHOW PROFILES 或 SHOW PROFILE 查看查询时间，判断是否因索引失效导致性能下降。

检查慢查询日志：开启慢查询日志，分析哪些查询因索引失效导致性能问题。

21.集合相关的接口以及实现类

Collection 接口体系

  1. List 接口
  • 特点：有序、可重复，允许索引访问。
  • 典型实现类
    • ArrayList：基于动态数组实现，适合随机访问（get(index)），但插入/删除效率较低。
    • LinkedList：基于双向链表实现，适合频繁插入/删除操作，但随机访问效率低。
    • Vector：线程安全版本，性能较 ArrayList 低，已被 ArrayList 替代。
    • Stack：继承自 Vector，提供栈操作（如 push()、pop()），但已不推荐使用。
  1. Set 接口
  • 特点：无序、不可重复。
  • 典型实现类
    • HashSet：基于哈希表实现，元素无序，依赖 hashCode() 和 equals() 方法判断重复。
    • LinkedHashSet：HashSet 的链表版本，维护插入顺序。
    • TreeSet：基于红黑树实现，元素自然排序或通过比较器排序。

Map 接口

  • 特点：键值对存储，键唯一，值可重复。
  • 典型实现类
    • HashMap：基于哈希表实现，键无序，允许一个 null 键和多个 null 值。
    • LinkedHashMap：HashMap 的链表版本，维护插入顺序或访问顺序。
    • TreeMap：基于红黑树实现，键自然排序或通过比较器排序。
    • Hashtable：线程安全版本，性能较低，已被 ConcurrentHashMap 替代。
    • ConcurrentHashMap：线程安全，支持高并发访问，性能优于 Hashtable。

26.Union和union all

UNION：合并多个查询结果集，并自动去除重复记录，最终返回的结果集是去重后的集合。由于需要去重，数据库引擎必须对结果集进行排序和比较，因此性能较低，尤其是当数据量较大时。

UNION ALL：合并多个查询结果集，保留所有记录（包括重复记录），最终返回的结果集是原始集合的拼接。直接拼接结果集，无需去重操作，性能更高，适合需要保留重复记录的场景。

27.事务传播

当事务方法被另一个事务方法调用时，事务应如何处理

传播行为 描述 案例
REQUIRED 如果当前存在事务，则加入该事务；如果当前没有事务，则创建一个新事务。 默认传播行为，适用于大多数场景。例如，用户下单时，订单服务和库存服务需要共享同一个事务。
REQUIRES_NEW 总是创建一个新事务，并挂起当前事务（如果存在）。 例如，记录日志时，日志服务需要独立于主业务事务，确保日志始终被记录，即使主业务事务回滚。
SUPPORTS 如果当前存在事务，则加入该事务；如果当前没有事务，则以非事务方式执行。 适用于可选事务场景。例如，查询用户信息时，如果存在事务则加入，否则直接查询。
NOT_SUPPORTED 以非事务方式执行，并挂起当前事务（如果存在）。 适用于不需要事务的场景。例如，导出报表时，为了避免事务占用资源，可以选择非事务方式执行。
MANDATORY 如果当前存在事务，则加入该事务；如果当前没有事务，则抛出异常。 适用于必须依赖外部事务的场景。例如，转账操作中，扣款和入账必须共享同一个事务，否则会导致数据不一致。
NEVER 以非事务方式执行，如果当前存在事务，则抛出异常。 适用于绝对不能有事务的场景。例如，某些敏感操作（如审计日志）需要确保没有事务干扰。
NESTED 如果当前存在事务，则创建一个嵌套事务；如果当前没有事务，则行为与 REQUIRED 相同。嵌套事务可以独立回滚，而不会影响外部事务。

30.Mysql存储引擎以及介绍

存储引擎是数据库底层用于存储、检索和管理数据的关键组件。不同的存储引擎提供了不同的特性，适用于不同的应用场景。

-InnoDB

  • 默认引擎：从MySQL 5.5版本开始，InnoDB成为默认存储引擎。
  • 特点
    • 事务支持：支持ACID（原子性、一致性、隔离性、持久性）事务。
    • 行级锁定：提高并发性能，减少锁冲突。
    • 外键约束：支持外键，保证数据完整性。
    • 崩溃恢复：提供崩溃恢复能力，确保数据一致性。
    • MVCC（多版本并发控制）：支持高并发读写。
  • 适用场景
    • 需要事务支持的应用（如金融系统、电商系统）。
    • 高并发读写场景。
    • 需要数据完整性和一致性的应用。

-MyISAM

  • 特点

    • 非事务型：不支持事务和外键。
    • 表级锁定：读写时会对整个表加锁，并发性能较低。
    • 全文索引：支持全文索引，适合文本搜索。
    • 高读取性能：对于只读或读多写少的场景，性能优于InnoDB。

  • 适用场景

    • 读多写少的应用（如数据仓库、日志系统）。
    • 需要全文索引的场景。

-MEMORY

  • 特点
    • 内存存储：数据存储在内存中，读写速度极快。
    • 表级锁定：并发性能较低。
    • 数据易失性：服务器重启或崩溃时，数据会丢失。
    • 固定长度行：不支持可变长度字段（如VARCHAR）。
  • 适用场景
    • 临时数据存储（如缓存、会话数据）。
    • 需要快速读写的临时表。

-CSV

  • 特点
    • CSV文件存储：数据存储在CSV文件中，便于与其他系统交换数据。
    • 不支持索引：查询性能较低。
    • 简单易用：适合小型数据集。
  • 适用场景
    • 数据导入导出。
    • 与其他系统（如Excel）交互。

-ARCHIVE

  • 特点
    • 高压缩比：数据存储高度压缩，节省存储空间。
    • 只支持插入和选择：不支持更新和删除。
    • 高写入性能：适合日志归档。
  • 适用场景
    • 日志归档和历史数据存储。

-NDB（Cluster）

  • 特点
    • 分布式存储：支持分布式集群，提供高可用性和可扩展性。
    • 同步复制：数据在多个节点间同步复制，确保数据一致性。
    • 内存存储：数据主要存储在内存中，读写性能高。
  • 适用场景
    • 高可用性和可扩展性要求高的应用（如大型网站、在线游戏）。

-FEDERATED

  • 特点
    • 联邦存储：允许访问远程MySQL服务器上的表，就像访问本地表一样。
    • 数据不存储在本地：仅存储表的元数据。
  • 适用场景
    • 分布式数据库系统。
    • 跨数据库的数据集成。

-BLACKHOLE

  • 特点
    • 黑洞存储：写入的数据会被丢弃，不存储在任何地方。
    • 复制功能：常用于主从复制中的中继日志。
  • 适用场景
    • 数据复制的中继。
    • 测试和开发环境。

-MERGE

  • 特点
    • 表合并：将多个相同的MyISAM表合并为一个逻辑表。
    • 统一视图：对用户提供一个统一的视图。
  • 适用场景
    • 分区表管理。
    • 历史数据归档。

如何选择存储引擎？

  1. 事务支持：如果需要事务支持，选择InnoDB。
  2. 并发性能：高并发读写场景选择InnoDB，低并发读多写少场景选择MyISAM。
  3. 数据持久性：如果数据需要持久化，避免使用MEMORY引擎。
  4. 特殊需求
    • 全文索引：选择MyISAM。
    • 日志归档：选择ARCHIVE。
    • 分布式存储：选择NDB。

31.Redis多服务器怎么平均分布存储数据

Redis Cluster 是 Redis 官方提供的分布式解决方案，支持数据的自动分片和节点间的故障转移。

实现步骤：

哈希槽分配

  • Redis Cluster 将 16384 个哈希槽均匀分配到各个节点上。
  • 每个键通过 CRC16(key) % 16384 计算所属的哈希槽，然后路由到对应的节点。

节点管理

  • 使用 redis-cli --cluster create 命令创建集群，并指定节点数量。
  • Redis 会自动将哈希槽分配到各个节点，确保均匀分布。

数据迁移：当节点增减时，Redis 会自动进行哈希槽的重新分配和数据迁移，尽量减少对现有服务的影响。

客户端连接：客户端只需连接到集群中的任意一个节点，Redis 会自动路由请求到正确的节点。

使用客户端分片

客户端实现自定义的分片逻辑。

实现步骤：

一致性哈希算法

  • 使用一致性哈希算法将键映射到虚拟节点上，虚拟节点再映射到实际的 Redis 服务器。
  • 一致性哈希可以减少节点增减时的数据迁移量。

分片规则

  • 根据键的某种特征（如哈希值、前缀等）计算目标节点。
  • 例如，使用 hash(key) % N（N 为节点数量）计算目标节点。

客户端实现

  • 在客户端代码中实现分片逻辑，根据键选择对应的 Redis 服务器。

33.拉姆达表达式的stream的模式

  1. 创建 Stream：从数据源生成流。
  2. 中间操作：使用 Lambda 表达式对流中的数据进行处理。
    1. filter：筛选符合条件的元素。
    2. map：将元素转换为另一种形式。
    3. flatMap：将每个元素映射为一个流，并将所有流扁平化为一个流。
    4. sorted：对流中的元素进行排序。
    5. distinct：去重。
    6. limit：限制流的大小。
    7. skip：跳过前 n 个元素。
  3. 终止操作：触发流的执行并返回结果。
    1. forEach：对每个元素执行操作。
    2. collect：将流中的元素收集到集合中。
    3. reduce：将流中的元素归约为一个值。
    4. count：返回流中元素的数量。
    5. findFirst：返回第一个元素。
    6. anyMatch：判断是否存在至少一个元素符合条件。
    7. allMatch：判断是否所有元素都符合条件。
    8. noneMatch：判断是否没有元素符合条件。
  4. 并行处理：利用并行流提高性能。
List<Integer> numbers = Arrays.asList(1, 2, 3, 4, 5);
int sum = numbers.parallelStream()  // 使用并行流
                 .reduce(0, Integer::sum);  // 计算总和
System.out.println(sum);  // 输出 15
List<String> list = Arrays.asList("apple1", "banana2", "cherry3", "date4");
List<String> result = list.stream()
                          .filter(s -> s.matches(".*\\d"))  // 筛选包含数字的字符串
                          .map(s -> s.replaceAll("\\d", ""))  // 移除数字
                          .sorted()  // 排序
                          .collect(Collectors.toList());  // 收集到列表
System.out.println(result);  // 输出 [apple, banana, cherry, date]

36.Spring boot注解解释及特点

@SpringBootApplication：复合注解，包含：

  • @Configuration：允许在类中定义 @Bean 方法，将方法的返回值注册为 Spring 容器中的 Bean。
  • @EnableAutoConfiguration：根据项目的依赖自动配置 Spring 上下文，减少手动配置的工作量。
  • @ComponentScan：自动扫描并注册指定包及其子包中的组件（如 @Component、@Service、@Repository 等）。

​ @RestController ：@Controller 和 @ResponseBody 的组合注解，表示该类是一个 RESTful 控制器，方法的返回值会直接作为 HTTP 响应体返回，而不是视图名称。

  • 用于开发 RESTful API，返回的数据通常是 JSON 或 XML 格式。
  • 简化了开发，无需在每个方法上添加 @ResponseBody。

@RequestMapping：映射 HTTP 请求到处理器方法，可以指定请求路径、HTTP 方法、请求参数等。

  • 通用注解，可以用于类和方法上。
  • 支持多种配置属性，如 value（路径）、method（HTTP 方法）、params（请求参数）等。
  • 衍生出更具体的注解，如 @GetMapping、@PostMapping、@PutMapping、@DeleteMapping，分别用于处理不同的 HTTP 方法。

@Autowired：于自动装配 Bean，Spring 会根据类型（或名称，结合 @Qualifier）将依赖注入到目标对象中。

  • 可以用于字段、构造函数或方法上。
  • 默认按类型匹配，如果需要按名称匹配，可以结合 @Qualifier 使用。
  • 是 Spring 依赖注入的核心注解之一。

@Service：标注业务逻辑层的组件，是 @Component 的一个特化注解。

  • 标识该类是一个服务组件，Spring 会将其注册为 Bean。
  • 通常用于业务逻辑的实现类，便于区分不同层次的组件。

@Repository：标注数据访问层的组件，是 @Component 的一个特化注解。

  • 标识该类是一个数据访问组件（如 DAO），Spring 会将其注册为 Bean。
  • 结合 Spring 的异常转换机制，可以将数据访问层的异常转换为 Spring 的数据访问异常。

@Value：注入外部配置的值，如 application.properties 或 application.yml 中的属性。

  • 支持直接注入字符串、环境变量或占位符表达式。
  • 示例：@Value("${my.property}")。

@ConfigurationProperties：将配置文件中的属性绑定到一个 Java 对象上。

  • 需要配合 @Component 或 @EnableConfigurationProperties 使用。
  • 适合绑定一组相关的配置属性，便于管理和使用。

@Transactional：声明事务边界，支持方法或类级别的事务管理。

  • Spring 会自动管理事务的开启、提交和回滚。
  • 可以指定事务的传播行为、隔离级别等属性。

@Async：标记方法为异步执行，Spring 会将该方法提交到线程池中异步执行。

  • 需要启用异步支持（@EnableAsync）。
  • 返回类型可以是 void 或 Future。

@Scheduled：标记方法为定时任务，Spring 会按照指定的时间表达式执行该方法。

  • 需要启用定时任务支持（@EnableScheduling）。
  • 支持多种时间表达式，如 cron、fixedRate、fixedDelay。

@Conditional：用于条件化装配 Bean，只有满足指定条件时，Spring 才会创建该 Bean。

  • 可以结合其他条件注解（如 @ConditionalOnProperty、@ConditionalOnClass）使用。
  • 提供了灵活的 Bean 装配机制。

@ComponentScan

  • 解释：
    @ComponentScan 用于指定 Spring 应该扫描的包路径，自动发现并注册组件。
  • 特点
    • 通常与 @SpringBootApplication 一起使用，默认扫描启动类所在包及其子包。
    • 可以自定义扫描路径，排除特定组件等。

@EnableAutoConfiguration：通常通过 @SpringBootApplication 隐式使用，但 @EnableAutoConfiguration 单独使用时，可以启用 Spring Boot 的自动配置功能。

  • 根据类路径中的依赖自动配置 Spring 上下文。
  • 可以通过 exclude 属性排除特定的自动配置类。

@Entity：标注 JPA 实体类，表示该类与数据库表对应。

  • 需要结合 JPA 提供者（如 Hibernate）使用。
  • 通常与 @Table、@Id 等注解一起使用。

41.Mysql性能调优

优化数据库配置、查询语句、表结构、硬件资源等

核心配置参数优化

  1. 缓冲池（Buffer Pool）

    • 调整innodb_buffer_pool_size参数，通常设置为物理内存的70%-80%，确保大部分数据和索引驻留在内存中，减少磁盘I/O。
    • 设置innodb_buffer_pool_instances为8-16个实例，减少锁竞争。

  2. 日志与事务

    • 增大innodb_log_file_size至1-2G，减少日志checkpoint频率。

    • 调整

      innodb_flush_log_at_trx_commit

      参数：

      • 1：强一致性，适合金融等对数据安全性要求高的场景。
      • 2：性能优先，适合高并发场景。

    • 设置sync_binlog为1（主库）或0（从库），平衡性能与数据安全性。

  3. 连接与线程

    • 根据业务负载调整max_connections，避免过高导致内存耗尽。
    • 设置thread_cache_size为50-100，减少线程创建销毁开销。
    • 调整back_log为100-500，处理高并发时的连接积压。

  4. 查询缓存

    • 通常建议关闭查询缓存（query_cache_type=OFF），避免在高写场景下导致缓存失效。

二、查询优化

  1. 索引优化
    • 选择合适的索引字段：为高频查询字段创建索引，确保索引列具有足够的唯一性。
    • 利用复合索引：在多个列上创建复合索引，减少查询时的回表操作。
    • 定期维护索引：通过OPTIMIZE TABLE命令重新组织表结构和索引，避免碎片化。
  2. 查询语句优化
    • 避免全表扫描：使用索引限制扫描的行数，或拆分大查询为多个小查询。
    • 限制返回行数：使用LIMIT子句限制返回的数据量，减少传输的数据量。
    • 避免使用不必要的函数：在查询中尽量避免使用复杂的表达式和函数，防止索引失效。
    • 合理利用子查询与JOIN：优先选择关联性强、数据量较小的表进行联接，将子查询转换为JOIN形式以提高执行效率。
  3. EXPLAIN分析
    • 使用EXPLAIN关键字查看MySQL如何执行查询，关注type列（避免ALL全表扫描）和key列（确认是否命中索引）。

三、表结构优化

  1. 选择合适的数据类型
    • 使用INT代替VARCHAR存储数字，使用DATETIME代替VARCHAR存储时间。
    • 对于定长字段（如性别、状态标识），使用CHAR类型可提高性能。
  2. 避免冗余和过大字段
    • 大文本字段使用TEXT并独立成表，不在索引列上使用函数或运算。
    • 遵循最左前缀原则（联合索引按顺序使用），避免LIKE '%keyword%'左模糊查询。
  3. 分区表
    • 对于大型数据集，使用分区表将数据分布在不同的物理位置上，提高访问速度并简化管理。

四、硬件与架构优化

  1. 硬件升级
    • 提升CPU、内存、SSD磁盘等硬件性能，特别是使用SSD替代HDD可以显著降低随机I/O延迟。
    • 对于生产环境，可采用RAID 10来兼顾数据冗余和高性能。
  2. 读写分离
    • 主库处理写操作，从库处理读操作，使用中间件（如ProxySQL）或框架（Spring动态数据源）实现自动路由。
  3. 分库分表
    • 按业务拆分数据库（垂直分库）或按哈希或范围拆分表（水平分表），使用ShardingSphere、Vitess等工具实现。

五、监控与诊断

  1. 慢查询日志
    • 开启慢查询日志（slow_query_log=ON），设置long_query_time为1秒，使用mysqldumpslow或pt-query-digest工具分析慢查询。
  2. 性能监控
    • 使用Prometheus + Grafana + mysqld_exporter进行性能监控，通过SHOW PROCESSLIST、SHOW ENGINE INNODB STATUS等命令诊断锁等待、文件I/O等问题。

六、其他优化技巧

  1. 使用缓存
    • 利用MySQL查询缓存或外部缓存（如Redis）减少数据库压力。
  2. 优化锁机制
    • 减少事务粒度，使用行级锁（InnoDB），避免长事务。
  3. 定期维护
    • 清理冗余数据，定期执行OPTIMIZE TABLE整理碎片，设置自动清理机制（如定时任务）。
  4. 分析慢查询日志
    • 通过慢日志定位低效SQL并针对性优化，使用EXPLAIN分析执行计划。

42.软件测试标准、目的

软件测试目的

  1. 缺陷发现：识别代码、逻辑或设计中的错误，预防上线后故障。
  2. 需求验证：确保软件符合功能与非功能需求（如性能、安全性）。
  3. 质量提升：通过持续测试优化稳定性、用户体验及兼容性。
  4. 风险控制：降低因软件缺陷导致的经济、安全或声誉损失。
  5. 合规保障：满足法规（GDPR、HIPAA）及行业标准（PCI-DSS）。
  6. 过程改进：分析缺陷根源，优化开发流程，减少未来错误。
  7. 性能评估：测试负载、压力下的表现，确保高并发或大数据处理能力。
  8. 用户信心建立：通过全面测试增强客户对产品质量的信任。

43.事务注解的参数

value 或 transactionManager

  • 作用：指定事务管理器。
  • 说明：在有多个事务管理器的情况下，用于选择使用哪个事务管理器。

propagation

  • 作用：定义事务的传播行为。

isolation

  • 作用：定义事务的隔离级别。

timeout

  • 作用：指定事务的超时时间（单位：秒）。
  • 说明：如果事务在指定时间内未完成，则回滚。

readOnly

  • 作用：指定事务是否为只读事务。
  • 说明
    • true：事务为只读，优化查询性能。
    • false（默认）：事务支持读写操作

rollbackFor

  • 作用：指定哪些异常会触发事务回滚。
  • 说明：可以指定异常类或异常类数组。

44.lamda表达式以及做了什么

用于创建匿名函数的一种语法结构。它允许开发者在代码中直接定义一个简短的函数，而无需显式地声明一个函数名称或将其定义为独立的函数体。Lambda 表达式通常用于需要函数作为参数的场景，例如集合操作、回调函数、事件处理等。

//排序列表
ublic static void main(String[] args) {
        List<String> names = Arrays.asList("Alice", "Bob", "Charlie");
        // 使用 Lambda 表达式对列表进行排序
        names.sort((s1, s2) -> s1.compareTo(s2));
        names.forEach(name -> System.out.println(name));
    }

//实现自定义比较器
public static void main(String[] args) {
        List<String> names = Arrays.asList("Alice", "Bob", "Charlie");
        // 使用 Lambda 表达式实现自定义比较器（按字符串长度排序）
        names.sort((s1, s2) -> Integer.compare(s1.length(), s2.length()));
        names.forEach(System.out::println); // 使用方法引用简化打印
    }

//过滤列表
public static void main(String[] args) {
        List<Integer> numbers = Arrays.asList(1, 2, 3, 4, 5, 6);
        // 使用 Lambda 表达式过滤出偶数
        List<Integer> evenNumbers = numbers.stream()
                                          .filter(n -> n % 2 == 0)
                                          .collect(Collectors.toList());
        evenNumbers.forEach(System.out::println);
    }

46.依赖注入

依赖注入是一种设计模式，用于实现控制反转（IoC, Inversion of Control）。它通过将对象的依赖关系从代码内部管理转移到外部容器中，使得对象更加松耦合、可测试和可维护。

构造函数注入（Constructor Injection）

  • 通过构造函数将依赖传递给对象。
  • 优点：依赖关系明确，不可变（Immutable）。

Setter 方法注入（Setter Injection）

  • 通过 Setter 方法设置依赖。
  • 优点：依赖可以在对象创建后动态修改。

接口注入（Interface Injection）

  • 通过实现一个接口来注入依赖。
  • 优点：依赖注入的契约明确。
  • 缺点：不常用，代码冗余。
